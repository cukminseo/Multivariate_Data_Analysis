##### 유의성 테스트 (z-test) #####

### z-점수 계산
mn = df2['diff'].mean()  # 'diff' 컬럼의 평균 계산
std = df2['diff'].std()  # 'diff' 컬럼의 표준 편차 계산
n = len(df2)  # 데이터 개수 계산
se = std / np.sqrt(n)  # 표준 오차 계산
z_score = mn / se  # z-점수 계산

# z-점수를 소수점 한 자리까지 반올림하여 출력
print("z-score:", z_score.round(1))

import numpy as np
import pandas as pd


# 성별에 따른 데이터프레임 분리
df_male = df[df['sex'] == 1]  # 'sex' 값이 1인 행(남성)을 선택하여 새로운 데이터프레임 생성
df_female = df[df['sex'] == 2]  # 'sex' 값이 2인 행(여성)을 선택하여 새로운 데이터프레임 생성

# 평균 소득 계산
mn_male = df_male['income'].mean()  # 남성의 평균 소득 계산
mn_female = df_female['income'].mean()  # 여성의 평균 소득 계산
mn_diff = mn_male - mn_female  # 남성과 여성의 평균 소득 차이 계산
print("mn_diff:", mn_diff)

# 표준편차 계산
std_male = df_male['income'].std()  # 남성의 소득 표준 편차 계산
std_female = df_female['income'].std()  # 여성의 소득 표준 편차 계산

# 표준오차 계산
se_male = std_male / np.sqrt(len(df_male))  # 남성 소득의 표준 오차 계산
se_female = std_female / np.sqrt(len(df_female))  # 여성 소득의 표준 오차 계산
se_tot = np.sqrt(se_male**2 + se_female**2)  # 두 그룹의 표준 오차를 합산하여 전체 표준 오차 계산
print('표준오차:', np.round(se_tot, 3))

# Z-스코어 계산
z_score = mn_diff / se_tot  # 평균 소득 차이를 전체 표준 오차로 나누어 Z-스코어 계산
print("Z-스코어:", round(z_score, 3))
